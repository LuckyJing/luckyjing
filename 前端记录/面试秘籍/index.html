{

    if(!value)

    {

        if(obj.currentStyle)

        {

           return obj.currentStyle(attr)

        }

            obj.getComputedStyle(attr,false)

        }

        obj.style[attr]=value

    }

}

22.编写一个数组去重的方法
function oSort(arr)
{

    var result ={};

    var newArr=[];

    for(var i=0;i<arr.length;i++)

    {

        if(!result[arr[i]])

        {

            newArr.push(arr[i])

            result[arr[i]]=1

        }

    }

    return newArr

}
